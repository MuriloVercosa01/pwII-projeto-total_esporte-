CREATE DATABASE total_esporte;

use total_esporte;

CREATE TABLE cadastros(
	id_cliente int PRIMARY KEY AUTO_INCREMENT,
    email varchar(150),
    senha varchar(10),
    nome varchar(50)
);
CREATE TABLE cart(
    id_cart int PRIMARY KEY AUTO_INCREMENT,
    id_cliente int,
    status boolean,
    FOREIGN KEY (id_cliente) REFERENCES cadastros(id_cliente)
);
CREATE TABLE categoria(
    id_categoria int PRIMARY key AUTO_INCREMENT,
    prod_cate varchar(45)
);
CREATE TABLE produto(
    id_produto int PRIMARY KEY AUTO_INCREMENT,
    modelo varchar(50),
    preco double,
    tamanho int,
    imagem varchar(500),
    id_categoria int,
    FOREIGN KEY (id_categoria) REFERENCES categoria(id_categoria)
);
CREATE TABLE cart_itens(
    id_cart_itens int PRIMARY KEY AUTO_INCREMENT,
    id_cart INT,
    id_cliente INT,
    id_produto INT,
    FOREIGN KEY (id_cliente) REFERENCES cadastros (id_cliente),
    FOREIGN KEY (id_cart) REFERENCES cart (id_cart),
    FOREIGN KEY (id_produto) REFERENCES produto (id_produto),
    quantidade int
);

SELECT
produto.id_produto,
    produto.modelo,
    categoria.categoria_prod,
    produto.desc_breve,
    produto.valor
FROM
	produto
JOIN
	categoria ON produto.categoria = categoria.id_cate;
